93  template <typename F>
94  void forEach(F&& onCountersMetadata) const
95  {
96  std::int32_t id = 0;
97 
98  for (util::index_t i = 0, capacity = m_metadataBuffer.capacity(); i < capacity; i += METADATA_LENGTH)
99  {
100  std::int32_t recordStatus = m_metadataBuffer.getInt32Volatile(i);
101 
102  if (RECORD_UNUSED == recordStatus)
103  {
104  break;
105  }
106  else if (RECORD_ALLOCATED == recordStatus)
107  {
108  const struct CounterMetaDataDefn& record = m_metadataBuffer.overlayStruct<CounterMetaDataDefn>(i);
109 
110  const std::string label = m_metadataBuffer.getString(i + LABEL_LENGTH_OFFSET);
111  const AtomicBuffer keyBuffer(m_metadataBuffer.buffer() + i + KEY_OFFSET, sizeof(CounterMetaDataDefn::key));
112 
113  onCountersMetadata(id, record.typeId, keyBuffer, label);
114  }
115 
116  id++;
117  }
118  }
