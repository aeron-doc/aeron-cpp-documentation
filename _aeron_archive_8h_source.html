403  template <typename RecordingDescriptorConsumer>
404  std::int32_t pollForDescriptors(
405  const std::int64_t correlationId, const std::int32_t recordCount, RecordingDescriptorConsumer&& consumer)
406  {
407  const auto deadlineTs = std::chrono::high_resolution_clock::now() + m_messageTimeout;
408  auto& poller = m_recordingDescriptorPoller;
409  poller.reset(correlationId, recordCount, consumer);
410 
411  while (true)
412  {
413  const int fragments = poller.poll();
414 
415  if (poller.isDispatchComplete())
416  {
417  return recordCount - poller.remainingRecordCount();
418  }
419 
420  m_aeron.conductorAgentInvoker().invoke();
421 
422  if (fragments > 0)
423  {
424  continue;
425  }
426 
427  if (!poller.subscription().isConnected())
428  {
429  throw util::IllegalStateException("subscription to archive is not connected", SOURCEINFO);
430  }
431 
432  if (std::chrono::high_resolution_clock::now() > deadlineTs)
433  {
434  throw util::ConductorServiceTimeoutException(
435  "awaiting recording descriptors: correlationId=" + std::to_string(correlationId), SOURCEINFO);
436  }
437 
438  m_idleStrategy.idle();
439  }
