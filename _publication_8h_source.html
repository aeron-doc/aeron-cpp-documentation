267  inline std::int64_t offer(
268  concurrent::AtomicBuffer& buffer,
269  util::index_t offset,
270  util::index_t length,
271  const on_reserved_value_supplier_t& reservedValueSupplier)
272  {
273  std::int64_t newPosition = PUBLICATION_CLOSED;
274 
275  if (!isClosed())
276  {
277  const std::int64_t limit = m_publicationLimit.getVolatile();
278  const std::int32_t termCount = LogBufferDescriptor::activeTermCount(m_logMetaDataBuffer);
279  TermAppender *termAppender = m_appenders[LogBufferDescriptor::indexByTermCount(termCount)].get();
280  const std::int64_t rawTail = termAppender->rawTailVolatile();
281  const std::int64_t termOffset = rawTail & 0xFFFFFFFF;
282  const std::int32_t termId = LogBufferDescriptor::termId(rawTail);
283  const std::int64_t position =
284  LogBufferDescriptor::computeTermBeginPosition(
285  termId, m_positionBitsToShift, m_initialTermId) + termOffset;
286 
287  if (termCount != (termId - m_initialTermId))
288  {
289  return ADMIN_ACTION;
290  }
291 
292  if (position < limit)
293  {
294  std::int32_t resultingOffset;
295  if (length <= m_maxPayloadLength)
296  {
297  resultingOffset = termAppender->appendUnfragmentedMessage(
298  m_headerWriter, buffer, offset, length, reservedValueSupplier, termId);
299  }
300  else
301  {
302  checkForMaxMessageLength(length);
303  resultingOffset = termAppender->appendFragmentedMessage(
304  m_headerWriter, buffer, offset, length, m_maxPayloadLength, reservedValueSupplier, termId);
305  }
306 
307  newPosition =
308  Publication::newPosition(
309  termCount, static_cast<std::int32_t>(termOffset), termId, position, resultingOffset);
310  }
311  else
312  {
313  newPosition = Publication::backPressureStatus(position, length);
314  }
315  }
316 
317  return newPosition;
318  }
