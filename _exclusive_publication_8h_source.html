264  inline std::int64_t offer(
265  concurrent::AtomicBuffer& buffer,
266  util::index_t offset,
267  util::index_t length,
268  const on_reserved_value_supplier_t& reservedValueSupplier)
269  {
270  std::int64_t newPosition = PUBLICATION_CLOSED;
271 
272  if (!isClosed())
273  {
274  const std::int64_t limit = m_publicationLimit.getVolatile();
275  ExclusiveTermAppender *termAppender = m_appenders[m_activePartitionIndex].get();
276  const std::int64_t position = m_termBeginPosition + m_termOffset;
277 
278  if (position < limit)
279  {
280  std::int32_t result;
281  if (length <= m_maxPayloadLength)
282  {
283  result = termAppender->appendUnfragmentedMessage(
284  m_termId,
285  m_termOffset,
286  m_headerWriter,
287  buffer,
288  offset,
289  length,
290  reservedValueSupplier);
291  }
292  else
293  {
294  checkForMaxMessageLength(length);
295  result = termAppender->appendFragmentedMessage(
296  m_termId,
297  m_termOffset,
298  m_headerWriter,
299  buffer,
300  offset,
301  length,
302  m_maxPayloadLength,
303  reservedValueSupplier);
304  }
305 
306  newPosition = ExclusivePublication::newPosition(result);
307  }
308  else
309  {
310  newPosition = ExclusivePublication::backPressureStatus(position, length);
311  }
312  }
313 
314  return newPosition;

352  BufferIterator startBuffer,
353  BufferIterator lastBuffer,
354  const on_reserved_value_supplier_t& reservedValueSupplier = DEFAULT_RESERVED_VALUE_SUPPLIER)
355  {
356  util::index_t length = 0;
357  for (BufferIterator it = startBuffer; it != lastBuffer; ++it)
358  {
359  if (AERON_COND_EXPECT(length + it->capacity() < 0, false))
360  {
361  throw aeron::util::IllegalStateException(
362  aeron::util::strPrintf("length overflow: %d + %d -> %d",
363  length,
364  it->capacity(),

366  SOURCEINFO);
367  }
368 
369  length += it->capacity();
370  }

563  inline std::int64_t newPosition(const std::int32_t resultingOffset)
564  {
565  if (resultingOffset > 0)
566  {
567  m_termOffset = resultingOffset;
568 
569  return m_termBeginPosition + resultingOffset;
570  }
571 
572  if ((m_termBeginPosition + termBufferLength()) >= m_maxPossiblePosition)
573  {
574  return MAX_POSITION_EXCEEDED;
575  }
576 
577  const int nextIndex = LogBufferDescriptor::nextPartitionIndex(m_activePartitionIndex);
578  const std::int32_t nextTermId = m_termId + 1;
579 
580  m_activePartitionIndex = nextIndex;
581  m_termOffset = 0;
582  m_termId = nextTermId;
583  m_termBeginPosition =
584  LogBufferDescriptor::computeTermBeginPosition(nextTermId, m_positionBitsToShift, m_initialTermId);
585 
586  const std::int32_t termCount = nextTermId - m_initialTermId;
587 
588  LogBufferDescriptor::initializeTailWithTermId(m_logMetaDataBuffer, nextIndex, nextTermId);
589  LogBufferDescriptor::activeTermCountOrdered(m_logMetaDataBuffer, termCount);
590 
591  return ADMIN_ACTION;
592  }
