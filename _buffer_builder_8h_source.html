88  inline static std::uint32_t findSuitableCapacity(std::uint32_t currentCapacity, std::uint32_t requiredCapacity)
89  {
90  std::uint32_t capacity = currentCapacity;
91 
92  do
93  {
94  const std::uint32_t newCapacity = capacity + (capacity >> 1);
95 
96  if (newCapacity < capacity || newCapacity > BUFFER_BUILDER_MAX_CAPACITY)
97  {
98  if (capacity == BUFFER_BUILDER_MAX_CAPACITY)
99  {
100  throw util::IllegalStateException(
101  util::strPrintf("Max capacity reached: %d", BUFFER_BUILDER_MAX_CAPACITY), SOURCEINFO);
102  }
103 
104  capacity = BUFFER_BUILDER_MAX_CAPACITY;
105  }
106  else
107  {
108  capacity = newCapacity;
109  }
110  }
111  while (capacity < requiredCapacity);
112 
113  return capacity;
114  }
115 
116  void ensureCapacity(std::uint32_t additionalCapacity)
117  {
118  const std::uint32_t requiredCapacity = m_limit + additionalCapacity;
119 
120  if (requiredCapacity > m_capacity)
121  {
122  const std::uint32_t newCapacity = findSuitableCapacity(m_capacity, requiredCapacity);
123  std::unique_ptr < std::uint8_t[] > newBuffer(new std::uint8_t[newCapacity]);
124 
125  ::memcpy(&newBuffer[0], &m_buffer[0], m_limit);
126  m_buffer = std::move(newBuffer);
127  m_capacity = newCapacity;
128  }
129  }
