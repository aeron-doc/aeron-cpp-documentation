52  void transmit(std::int32_t msgTypeId, concurrent::AtomicBuffer& srcBuffer, util::index_t srcIndex, util::index_t length)
53  {
54  RecordDescriptor::checkMsgTypeId(msgTypeId);
55  checkMessageLength(length);
56 
57  std::int64_t currentTail = m_buffer.getInt64(m_tailCounterIndex);
58  std::int32_t recordOffset = (std::int32_t) currentTail & m_mask;
59  const std::int32_t recordLength = length + RecordDescriptor::HEADER_LENGTH;
60  const std::int32_t alignedRecordLength = util::BitUtil::align(recordLength, RecordDescriptor::RECORD_ALIGNMENT);
61  const std::int64_t newTail = currentTail + alignedRecordLength;
62  const std::int32_t toEndOfBuffer = m_capacity - recordOffset;
63 
64  if (toEndOfBuffer < alignedRecordLength)
65  {
66  signalTailIntent(m_buffer, newTail + toEndOfBuffer);
67 
68  insertPaddingRecord(m_buffer, recordOffset, toEndOfBuffer);
69 
70  currentTail += toEndOfBuffer;
71  recordOffset = 0;
72  }
73  else
74  {
75  signalTailIntent(m_buffer, newTail);
76  }
77 
78  m_buffer.putInt32(RecordDescriptor::lengthOffset(recordOffset), recordLength);
79  m_buffer.putInt32(RecordDescriptor::typeOffset(recordOffset), msgTypeId);
80 
81  m_buffer.putBytes(RecordDescriptor::msgOffset(recordOffset), srcBuffer, srcIndex, length);
82 
83  m_buffer.putInt64(m_latestCounterIndex, currentTail);
84  m_buffer.putInt64Ordered(m_tailCounterIndex, currentTail + alignedRecordLength);
85  }
86 
87 private:
88  AtomicBuffer& m_buffer;
89  util::index_t m_capacity;
90  util::index_t m_mask;
91  util::index_t m_maxMsgLength;
92  util::index_t m_tailIntentCounterIndex;
93  util::index_t m_tailCounterIndex;
94  util::index_t m_latestCounterIndex;

96  inline void checkMessageLength(util::index_t length)
97  {
98  if (length > m_maxMsgLength)
99  {
100  throw util::IllegalArgumentException(
101  util::strPrintf("encoded message exceeds maxMsgLength of %d, length=%d", m_maxMsgLength, length), SOURCEINFO);
102  }
103  }
104 
105  inline void signalTailIntent(AtomicBuffer& buffer, std::int64_t newTail)
106  {
107  buffer.putInt64Ordered(m_tailIntentCounterIndex, newTail);
108  // store fence = release()
109  atomic::release();
110  }
111 
112  inline static void insertPaddingRecord(AtomicBuffer& buffer, std::int32_t recordOffset, std::int32_t length)
113  {
114  buffer.putInt32(RecordDescriptor::lengthOffset(recordOffset), length);
115  buffer.putInt32(RecordDescriptor::typeOffset(recordOffset), RecordDescriptor::PADDING_MSG_TYPE_ID);
116  }
