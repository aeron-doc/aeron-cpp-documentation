39 public:

46  int receiveMessages()
47  {
48  return m_broadcastReceiver.receive(
49  [&](std::int32_t msgTypeId, AtomicBuffer& buffer, util::index_t offset, util::index_t length)
50  {
51  switch (msgTypeId)
52  {
53  case ControlProtocolEvents::ON_PUBLICATION_READY:
54  {
55  const PublicationBuffersReadyFlyweight publicationReady(buffer, offset);
56 
57  m_driverListener.onNewPublication(
58  publicationReady.streamId(),
59  publicationReady.sessionId(),
60  publicationReady.positionLimitCounterId(),
61  publicationReady.channelStatusIndicatorId(),
62  publicationReady.logFileName(),
63  publicationReady.correlationId(),
64  publicationReady.registrationId());
65  break;
66  }
67 
68  case ControlProtocolEvents::ON_EXCLUSIVE_PUBLICATION_READY:
69  {
70  const PublicationBuffersReadyFlyweight publicationReady(buffer, offset);
71 
72  m_driverListener.onNewExclusivePublication(
73  publicationReady.streamId(),
74  publicationReady.sessionId(),
75  publicationReady.positionLimitCounterId(),
76  publicationReady.channelStatusIndicatorId(),
77  publicationReady.logFileName(),
78  publicationReady.correlationId(),
79  publicationReady.registrationId());
80  break;
81  }
82 
83  case ControlProtocolEvents::ON_SUBSCRIPTION_READY:
84  {
85  const SubscriptionReadyFlyweight subscriptionReady(buffer, offset);
86 
87  m_driverListener.onSubscriptionReady(
88  subscriptionReady.correlationId(),
89  subscriptionReady.channelStatusIndicatorId());
90  break;
91  }
92 
93  case ControlProtocolEvents::ON_AVAILABLE_IMAGE:
94  {
95  const ImageBuffersReadyFlyweight imageReady(buffer, offset);
96 
97  m_driverListener.onAvailableImage(
98  imageReady.streamId(),
99  imageReady.sessionId(),
100  imageReady.logFileName(),
101  imageReady.sourceIdentity(),
102  imageReady.subscriberPositionId(),
103  imageReady.subscriberRegistrationId(),
104  imageReady.correlationId());
105  break;
106  }
107 
108  case ControlProtocolEvents::ON_OPERATION_SUCCESS:
109  {
110  const OperationSucceededFlyweight operationSucceeded(buffer, offset);
111 
112  m_driverListener.onOperationSuccess(operationSucceeded.correlationId());
113  break;
114  }
115 
116  case ControlProtocolEvents::ON_UNAVAILABLE_IMAGE:
117  {
118  const ImageMessageFlyweight imageMessage(buffer, offset);
119 
120  m_driverListener.onUnavailableImage(
121  imageMessage.streamId(),
122  imageMessage.correlationId(),
123  imageMessage.subscriptionRegistrationId());
124  break;
125  }
126 
127  case ControlProtocolEvents::ON_ERROR:
128  {
129  const ErrorResponseFlyweight errorResponse(buffer, offset);
130 
131  m_driverListener.onErrorResponse(
132  errorResponse.offendingCommandCorrelationId(),
133  errorResponse.errorCode(),
134  errorResponse.errorMessage());
135  break;
136  }
137 
138  case ControlProtocolEvents::ON_COUNTER_READY:
139  {
140  const CounterUpdateFlyweight response(buffer, offset);
141 
142  m_driverListener.onAvailableCounter(response.correlationId(), response.counterId());
143  break;
144  }
145 
146  case ControlProtocolEvents::ON_UNAVAILABLE_COUNTER:
147  {
148  const CounterUpdateFlyweight response(buffer, offset);
149 
150  m_driverListener.onUnavailableCounter(response.correlationId(), response.counterId());
151  break;
152  }
153 
154  default:
155  break;
156  }
157  });
158  }
159
