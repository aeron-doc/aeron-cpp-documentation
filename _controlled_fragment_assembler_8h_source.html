86  ControlledPollAction onFragment(AtomicBuffer& buffer, util::index_t offset, util::index_t length, Header& header)
87  {
88  const std::uint8_t flags = header.flags();
89  ControlledPollAction action = ControlledPollAction::CONTINUE;
90 
91  if ((flags & FrameDescriptor::UNFRAGMENTED) == FrameDescriptor::UNFRAGMENTED)
92  {
93  action = m_delegate(buffer, offset, length, header);
94  }
95  else
96  {
97  if ((flags & FrameDescriptor::BEGIN_FRAG) == FrameDescriptor::BEGIN_FRAG)
98  {
99  auto result = m_builderBySessionIdMap.emplace(header.sessionId(), static_cast<std::uint32_t>(m_initialBufferLength));
100  BufferBuilder& builder = result.first->second;
101 
102  builder
103  .reset()
104  .append(buffer, offset, length, header);
105  }
106  else
107  {
108  auto result = m_builderBySessionIdMap.find(header.sessionId());
109 
110  if (result != m_builderBySessionIdMap.end())
111  {
112  BufferBuilder& builder = result->second;
113  const std::uint32_t limit = builder.limit();
114 
115  if (builder.limit() != DataFrameHeader::LENGTH)
116  {
117  builder.append(buffer, offset, length, header);
118 
119  if ((flags & FrameDescriptor::END_FRAG) == FrameDescriptor::END_FRAG)
120  {
121  const util::index_t msgLength = builder.limit() - DataFrameHeader::LENGTH;
122  AtomicBuffer msgBuffer(builder.buffer(), builder.limit());
123  Header assemblyHeader(header);
124 
125  assemblyHeader.buffer(msgBuffer);
126  assemblyHeader.offset(0);
127  DataFrameHeader::DataFrameHeaderDefn& frame(
128  msgBuffer.overlayStruct<DataFrameHeader::DataFrameHeaderDefn>());
129 
130  frame.frameLength = DataFrameHeader::LENGTH + msgLength;
131  frame.sessionId = header.sessionId();
132  frame.streamId = header.streamId();
133  frame.termId = header.termId();
134  frame.flags = FrameDescriptor::UNFRAGMENTED;
135  frame.type = DataFrameHeader::HDR_TYPE_DATA;
136  frame.termOffset = header.termOffset() - (frame.frameLength - header.frameLength());
137 
138  action = m_delegate(msgBuffer, DataFrameHeader::LENGTH, msgLength, assemblyHeader);
139 
140  if (ControlledPollAction::ABORT == action)
141  {
142  builder.limit(limit);
143  }
144  else
145  {
146  builder.reset();
147  }
148  }
149  }
150  }
