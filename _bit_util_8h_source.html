81  * (Note: this only works on 32-bit types, when compiling with GCC
82  * or on newer x64 processors when compiling with Visual Studio.)
83  */
84  template<typename value_t>
85  inline int numberOfLeadingZeroes(value_t value) AERON_NOEXCEPT
86  {
87 #if defined(__GNUC__)
88  return __builtin_clz(value);
89 #elif defined(_MSC_VER)
90  unsigned long r;
91 
92  if (_BitScanReverse(&r, (unsigned long)value))

101  /* Taken from Hacker's Delight as ntz10 at http://www.hackersdelight.org/hdcodetxt/ntz.c.txt */
102  template<typename value_t>
103  inline int numberOfTrailingZeroes(value_t value) AERON_NOEXCEPT
104  {
105 #if defined(__GNUC__)
106  return __builtin_ctz(value);
107 #elif defined(_MSC_VER)
108  unsigned long r;
109 
110  if (_BitScanForward(&r, (unsigned long)value))
111  return r;
112 
113  return 32;
114 #else
115  static_assert(std::is_integral<value_t>::value, "numberOfTrailingZeroes only available on integral types");
116  static_assert(sizeof(value_t) <= 4, "numberOfTrailingZeroes only available on up to 32-bit integral types");

135  /*
136  * Finds the next power of 2 and returns it.
137  * Invalid arguments (negative, 0, or too large) always return 0 or min value of type.
138  */
139  template<typename value_t>
140  inline value_t findNextPowerOfTwo(value_t value) AERON_NOEXCEPT
141  {
142  static_assert(std::is_integral<value_t>::value, "findNextPowerOfTwo only available on integral types");
143 
144  value--;
145 
146  // Set all bits below the leading one using binary expansion http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
147  for (size_t i = 1; i < sizeof(value) * 8; i = i * 2)
148  value |= (value >> i);
149 
150  return value + 1;
151  }

155  * Solution is Figure 10-24.

157  template <typename value_t>
158  inline int fastMod3(value_t value) AERON_NOEXCEPT
159  {
160  static_assert(std::is_integral<value_t>::value, "fastMod3 only available on integral types");
161 
162  static char table[62] = {0,1,2, 0,1,2, 0,1,2, 0,1,2,
163  0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2,
164  0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2, 0,1,2,
165  0,1,2, 0,1,2, 0,1};
166 
167  value = (value >> 16) + (value & 0xFFFF); // Max 0x1FFFE.
168  value = (value >> 8) + (value & 0x00FF); // Max 0x2FD.
169  value = (value >> 4) + (value & 0x000F); // Max 0x3D.
170  return table[value];
