379  inline int controlledPoll(F&& fragmentHandler, int fragmentLimit)
380  {
381  int result = 0;
382 
383  if (!isClosed())
384  {
385  int fragmentsRead = 0;
386  std::int64_t initialPosition = m_subscriberPosition.get();
387  std::int32_t initialOffset = (std::int32_t) initialPosition & m_termLengthMask;
388  AtomicBuffer &termBuffer = m_termBuffers[LogBufferDescriptor::indexByPosition(initialPosition,
389  m_positionBitsToShift)];
390  std::int32_t resultingOffset = initialOffset;
391  const util::index_t capacity = termBuffer.capacity();
392 
393  m_header.buffer(termBuffer);
394 
395  try
396  {
397  do
398  {
399  const std::int32_t length = FrameDescriptor::frameLengthVolatile(termBuffer, resultingOffset);
400  if (length <= 0)
401  {
402  break;
403  }
404 
405  const std::int32_t frameOffset = resultingOffset;
406  const std::int32_t alignedLength = util::BitUtil::align(length, FrameDescriptor::FRAME_ALIGNMENT);
407  resultingOffset += alignedLength;
408 
409  if (FrameDescriptor::isPaddingFrame(termBuffer, frameOffset))
410  {
411  continue;
412  }
413 
414  m_header.offset(frameOffset);
415 
416  const ControlledPollAction action =
417  fragmentHandler(
418  termBuffer,
419  frameOffset + DataFrameHeader::LENGTH,
420  length - DataFrameHeader::LENGTH,
421  m_header);
422 
423  if (ControlledPollAction::ABORT == action)
424  {
425  resultingOffset -= alignedLength;
426  break;
427  }
428 
429  ++fragmentsRead;
430 
431  if (ControlledPollAction::BREAK == action)
432  {
433  break;
434  }
435  else if (ControlledPollAction::COMMIT == action)
436  {
437  initialPosition += (resultingOffset - initialOffset);
438  initialOffset = resultingOffset;
439  m_subscriberPosition.setOrdered(initialPosition);
440  }
441  }
442  while (fragmentsRead < fragmentLimit && resultingOffset < capacity);
443  }
444  catch (const std::exception& ex)
445  {
446  m_exceptionHandler(ex);
447  }
448 
449  const std::int64_t resultingPosition = initialPosition + (resultingOffset - initialOffset);
450  if (resultingPosition > initialPosition)
451  {
452  m_subscriberPosition.setOrdered(resultingPosition);
453  }
454 
455  result = fragmentsRead;
456  }

573  {

576  if (!isClosed())
577  {

582  std::int64_t position = initialPosition;

597  }
598

600  const std::int32_t alignedLength = util::BitUtil::align(length, FrameDescriptor::FRAME_ALIGNMENT);
601  offset += alignedLength;
602 
603  if (FrameDescriptor::isPaddingFrame(termBuffer, frameOffset))
604  {
605  continue;
606  }

609

611  fragmentHandler(

615  m_header);
