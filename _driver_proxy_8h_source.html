279  inline void writeCommandToDriver(const std::function<util::index_t(AtomicBuffer&, util::index_t &)>& filler)
280  {
281  AERON_DECL_ALIGNED(driver_proxy_command_buffer_t messageBuffer, 16);
282  AtomicBuffer buffer(messageBuffer);
283  util::index_t length = buffer.capacity();
284 
285  util::index_t msgTypeId = filler(buffer, length);
286 
287  if (!m_toDriverCommandBuffer.write(msgTypeId, buffer, 0, length))
288  {
289  throw util::IllegalStateException("couldn't write command to driver", SOURCEINFO);
290  }
