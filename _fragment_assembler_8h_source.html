85  inline void onFragment(AtomicBuffer& buffer, util::index_t offset, util::index_t length, Header& header)
86  {
87  const std::uint8_t flags = header.flags();
88 
89  if ((flags & FrameDescriptor::UNFRAGMENTED) == FrameDescriptor::UNFRAGMENTED)
90  {
91  m_delegate(buffer, offset, length, header);
92  }
93  else
94  {
95  if ((flags & FrameDescriptor::BEGIN_FRAG) == FrameDescriptor::BEGIN_FRAG)
96  {
97  auto result = m_builderBySessionIdMap.emplace(header.sessionId(), static_cast<std::uint32_t>(m_initialBufferLength));
98  BufferBuilder& builder = result.first->second;
99 
100  builder
101  .reset()
102  .append(buffer, offset, length, header);
103  }
104  else
105  {
106  auto result = m_builderBySessionIdMap.find(header.sessionId());
107 
108  if (result != m_builderBySessionIdMap.end())
109  {
110  BufferBuilder& builder = result->second;
111 
112  if (builder.limit() != DataFrameHeader::LENGTH)
113  {
114  builder.append(buffer, offset, length, header);
115 
116  if ((flags & FrameDescriptor::END_FRAG) == FrameDescriptor::END_FRAG)
117  {
118  const util::index_t msgLength = builder.limit() - DataFrameHeader::LENGTH;
119  AtomicBuffer msgBuffer(builder.buffer(), builder.limit());
120 
121  m_delegate(msgBuffer, DataFrameHeader::LENGTH, msgLength, header);
122 
123  builder.reset();
124  }
125  }
