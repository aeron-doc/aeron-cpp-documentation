153  inline std::int32_t nextCounterId()
154  {
155  const long long nowMs = m_clock();
156 
157  auto it = std::find_if(m_freeList.begin(), m_freeList.end(),
158  [&](std::int32_t counterId)
159  {
160  return
161  nowMs >=
162  m_metadataBuffer.getInt64Volatile(metadataOffset(counterId) + FREE_TO_REUSE_DEADLINE_OFFSET);
163  });
164 
165  if (it != m_freeList.end())
166  {
167  const std::int32_t counterId = *it;
168 
169  m_freeList.erase(it);
170 
171  m_valuesBuffer.putInt64Ordered(counterOffset(counterId), 0L);
172 
173  return counterId;
174  }
175 
176  return ++m_highWaterMark;
177  }
